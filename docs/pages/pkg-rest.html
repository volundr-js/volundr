@volundr/rest

REST API client - rate limits, retries, file uploads.

npm install @volundr/rest

Use this package standalone when you only need HTTP access to Discord's API.

Standalone usage

import { RestClient, DiscordAPIError, HTTPError } from "@volundr/rest";

const rest = new RestClient({ token: "..." });

// GET
const user = await rest.get("/users/123456");

// POST
await rest.post("/channels/123/messages", {
    body: { content: "Hello!" },
});

// File upload
await rest.post("/channels/123/messages", {
    body: { content: "Here's a file" },
    files: [{ name: "image.png", data: buffer }],
});

// Audit log reason
await rest.delete("/channels/123/messages/456", {
    reason: "Spam",
});

PropertyTypeDescription

try {
    await rest.post("/channels/123/messages", {
        body: { content: "" },
    });
} catch (err) {
    if (err instanceof DiscordAPIError) {
        console.log(err.code);    // 50006
        console.log(err.message); // "Cannot send an empty message"
    }
}
